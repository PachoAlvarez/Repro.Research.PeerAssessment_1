summaryComparison <- function(vector1, vector2) {
    a <- summary(vector1)
    namesT <- names(a)
    a <- as.vector(a)
    b <- as.vector(summary(vector2))
    summaryMeans <- data.frame(a, b)
    rownames(summaryMeans) <- namesT
    colnames(summaryMeans) <- c("Vector 1", "Vector 2")
    summaryMeans
}
